sautMax <- function(hclust_obj) {
  dend <- as.dendrogram(hclust_obj)
  # heights <- attr(dend, "height")
  heights <- get_branches_heights(dend)
  max_jump <- max(heights)
  print(max_jump)
  return(max_jump)
}
